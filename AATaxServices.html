<script>

  const agree = document.getElementById('agree');

  const submitBtn = document.getElementById('submitBtn');

  const agreeWarn = document.getElementById('agreeWarn');

  const toast = document.getElementById('toast');

  const form = document.getElementById('requestForm');


  // Enable button only when agreement is checked

  function updateButtonState(){

    submitBtn.disabled = !agree.checked;

    agreeWarn.style.display = agree.checked ? 'none' : 'block';

  }


  agree.addEventListener('change', () => {

    updateButtonState();

  });


  // Prevent submit unless agreed

  form.addEventListener('submit', (e) => {

    e.preventDefault();

    if (!agree.checked){

      updateButtonState();

      return;

    }


    // Demo behavior: show confirmation on-page (no backend)

    toast.classList.add('show');


    // Optional: clear fields

    form.reset();

    submitBtn.disabled = true;

    agreeWarn.style.display = 'none';


    // Hide toast after a bit

    setTimeout(() => toast.classList.remove('show'), 6000);

  });


  document.getElementById('year').textContent = new Date().getFullYear();
